import React, { useState, useEffect } from "react";
import { Inenting, Kamp, Plaas } from "@/entities/all";
import { Button } from "@/components/ui/button";
import { Plus } from "lucide-react";
import { AnimatePresence } from "framer-motion";

import InentingForm from "../components/inentings/InentingForm";
import InentingLys from "../components/inentings/InentingLys";
import InentingKalender from "../components/inentings/InentingKalender";

export default function InentingsPage() {
  const [inentings, setInentings] = useState([]);
  const [kampe, setKampe] = useState([]);
  const [plase, setPlase] = useState([]);
  const [showForm, setShowForm] = useState(false);
  const [editingInenting, setEditingInenting] = useState(null);

  useEffect(() => {
    loadData();
  }, []);

  const loadData = async () => {
    const [inentingData, kampData, plaasData] = await Promise.all([
      Inenting.list("-datum"),
      Kamp.list(),
      Plaas.list()
    ]);
    setInentings(inentingData);
    setKampe(kampData);
    setPlase(plaasData);
  };

  const handleSubmit = async (inentingData) => {
    if (editingInenting) {
      await Inenting.update(editingInenting.id, inentingData);
    } else {
      await Inenting.create(inentingData);
    }
    setShowForm(false);
    setEditingInenting(null);
    loadData();
  };

  const handleDelete = async (inenting) => {
    await Inenting.delete(inenting.id);
    loadData();
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-green-50 via-amber-50 to-orange-50 p-4 md:p-8">
      <div className="max-w-7xl mx-auto">
        <div className="flex flex-col md:flex-row justify-between items-start md:items-center mb-8 gap-4">
          <div>
            <h1 className="text-3xl md:text-4xl font-bold text-stone-800 mb-2">
              Inentings Program
            </h1>
            <p className="text-stone-600 text-lg">
              Hou rekord van alle inentings en volgende datums
            </p>
          </div>
          <Button 
            onClick={() => setShowForm(true)}
            className="w-full md:w-auto bg-green-600 hover:bg-green-700 shadow-lg"
          >
            <Plus className="w-5 h-5 mr-2" />
            Nuwe Inenting
          </Button>
        </div>

        <AnimatePresence>
          {showForm && (
            <InentingForm
              inenting={editingInenting}
              kampe={kampe}
              plase={plase}
              onSubmit={handleSubmit}
              onCancel={() => {
                setShowForm(false);
                setEditingInenting(null);
              }}
            />
          )}
        </AnimatePresence>

        <div className="grid lg:grid-cols-4 gap-6">
          <div className="lg:col-span-1">
            <InentingKalender 
              inentings={inentings}
              kampe={kampe}
            />
          </div>
          
          <div className="lg:col-span-3">
            <InentingLys 
              inentings={inentings}
              kampe={kampe}
              plase={plase}
              onEdit={(inenting) => {
                setEditingInenting(inenting);
                setShowForm(true);
              }}
              onDelete={handleDelete}
            />
          </div>
        </div>
      </div>
    </div>
  );
