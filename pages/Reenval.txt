import React, { useState, useEffect } from "react";
import { Reenval, Plaas } from "@/entities/all";
import { Button } from "@/components/ui/button";
import { Plus } from "lucide-react";
import { AnimatePresence } from "framer-motion";

import ReenvalForm from "../components/reenval/ReenvalForm";
import ReenvalStats from "../components/reenval/ReenvalStats";
import ReenvalLys from "../components/reenval/ReenvalLys";
import ReenvalGrafiek from "../components/reenval/ReenvalGrafiek"; // New Import

export default function ReenvalPage() {
  const [reenvals, setReenvals] = useState([]);
  const [plase, setPlase] = useState([]);
  const [showForm, setShowForm] = useState(false);
  const [editingReenval, setEditingReenval] = useState(null);

  useEffect(() => {
    loadData();
  }, []);

  const loadData = async () => {
    const [reenvalData, plaasData] = await Promise.all([
      Reenval.list("-datum"),
      Plaas.list()
    ]);
    setReenvals(reenvalData);
    setPlase(plaasData);
  };

  const handleSubmit = async (reenvalData) => {
    if (editingReenval) {
      await Reenval.update(editingReenval.id, reenvalData);
    } else {
      await Reenval.create(reenvalData);
    }
    setShowForm(false);
    setEditingReenval(null);
    loadData();
  };

  const handleDelete = async (reenval) => {
    await Reenval.delete(reenval.id);
    loadData();
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-green-50 via-amber-50 to-orange-50 p-4 md:p-8">
      <div className="max-w-7xl mx-auto">
        <div className="flex flex-col md:flex-row justify-between items-start md:items-center mb-8 gap-4">
          <div>
            <h1 className="text-3xl md:text-4xl font-bold text-stone-800 mb-2">
              Reenval Rekords
            </h1>
            <p className="text-stone-600 text-lg">
              Hou rekord van reenval vir al jou plase
            </p>
          </div>
          <Button 
            onClick={() => setShowForm(true)}
            className="w-full md:w-auto bg-blue-600 hover:bg-blue-700 shadow-lg"
          >
            <Plus className="w-5 h-5 mr-2" />
            Nuwe Reenval Entry
          </Button>
        </div>

        <AnimatePresence>
          {showForm && (
            <ReenvalForm
              reenval={editingReenval}
              plase={plase}
              onSubmit={handleSubmit}
              onCancel={() => {
                setShowForm(false);
                setEditingReenval(null);
              }}
            />
          )}
        </AnimatePresence>
        
        <div className="space-y-6">
          <ReenvalStats 
            reenvals={reenvals}
            plase={plase}
          />
          <ReenvalGrafiek 
            reenvals={reenvals}
            plase={plase}
          />
          <ReenvalLys 
            reenvals={reenvals}
            plase={plase}
            onEdit={(reenval) => {
              setEditingReenval(reenval);
              setShowForm(true);
            }}
            onDelete={handleDelete}
          />
        </div>
      </div>
    </div>
  );
}
