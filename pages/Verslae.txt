import React, { useState, useEffect } from "react";
import { Plaas, Kamp, Telling, Bemarking, Inenting, Reenval, Voerkral } from "@/entities/all";
import { Button } from "@/components/ui/button";
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { BarChart3, Download, FileText, FileSpreadsheet, Calendar } from "lucide-react";

import VerslagStats from "../components/verslae/VerslagStats";
import DataExport from "../components/verslae/DataExport";
import HistoriesVergelyking from "../components/verslae/HistoriesVergelyking";

export default function VerslaePage() {
  const [allData, setAllData] = useState({
    plase: [],
    kampe: [],
    tellings: [],
    bemarkings: [],
    inentings: [],
    reenvals: [],
    voerkrals: []
  });
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    loadAllData();
  }, []);

  const loadAllData = async () => {
    setLoading(true);
    try {
      const [plase, kampe, tellings, bemarkings, inentings, reenvals, voerkrals] = await Promise.all([
        Plaas.list(),
        Kamp.list(),
        Telling.list("-datum"),
        Bemarking.list("-datum"),
        Inenting.list("-datum"),
        Reenval.list("-datum"),
        Voerkral.list("-datum")
      ]);
      
      setAllData({
        plase,
        kampe,
        tellings,
        bemarkings,
        inentings,
        reenvals,
        voerkrals
      });
    } catch (error) {
      console.error("Fout met data laai:", error);
    }
    setLoading(false);
  };

  return (
    <div className="min-h-screen bg-gradient-to-br from-green-50 via-amber-50 to-orange-50 p-4 md:p-8">
      <div className="max-w-7xl mx-auto">
        <div className="flex flex-col md:flex-row justify-between items-start md:items-center mb-8 gap-4">
          <div>
            <h1 className="text-3xl md:text-4xl font-bold text-stone-800 mb-2">
              Verslae & Statistieke
            </h1>
            <p className="text-stone-600 text-lg">
              Omvattende verslae en data eksport opsies
            </p>
          </div>
        </div>

        <div className="grid lg:grid-cols-3 gap-6">
          <div className="lg:col-span-2 space-y-6">
            <VerslagStats 
              data={allData}
              loading={loading}
            />
            
            <HistoriesVergelyking 
              data={allData}
              loading={loading}
            />
          </div>
          
          <div>
            <DataExport 
              data={allData}
              loading={loading}
            />
          </div>
        </div>
      </div>
    </div>
  );
}
