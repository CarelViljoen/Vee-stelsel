import React from "react";
import { Card, CardHeader, CardTitle, CardContent } from "@/components/ui/card";
import { Table, TableBody, TableCell, TableHead, TableHeader, TableRow } from "@/components/ui/table";
import { Button } from "@/components/ui/button";
import { Badge } from "@/components/ui/badge";
import { Cloud, Edit, Trash2, Calendar } from "lucide-react";
import { format } from "date-fns";
import { af } from "date-fns/locale";

const tipeKleure = {
  lig: "bg-blue-100 text-blue-800",
  matig: "bg-green-100 text-green-800",
  swaar: "bg-yellow-100 text-yellow-800",
  storm: "bg-red-100 text-red-800"
};

export default function ReenvalLys({ reenvals, plase, onEdit, onDelete }) {
  const getPlaasNaam = (plaasId) => {
    const plaas = plase.find(p => p.id === plaasId);
    return plaas?.naam || 'Onbekende Plaas';
  };

  return (
    <Card className="shadow-lg bg-white/80 backdrop-blur-sm">
      <CardHeader className="p-6 border-b border-stone-100">
        <CardTitle className="flex items-center gap-2 text-xl font-bold text-stone-800">
          <Cloud className="w-5 h-5 text-blue-500" />
          Reenval Rekords
        </CardTitle>
      </CardHeader>
      <CardContent className="p-0">
        {reenvals.length > 0 ? (
          <div className="overflow-x-auto">
            <Table>
              <TableHeader>
                <TableRow className="bg-stone-50">
                  <TableHead className="font-semibold">Plaas</TableHead>
                  <TableHead className="font-semibold">Datum</TableHead>
                  <TableHead className="font-semibold">Reenval</TableHead>
                  <TableHead className="font-semibold">Tipe</TableHead>
                  <TableHead className="font-semibold">Aksies</TableHead>
                </TableRow>
              </TableHeader>
              <TableBody>
                {reenvals.map((reenval) => (
                  <TableRow key={reenval.id} className="hover:bg-stone-50 transition-colors">
                    <TableCell>
                      <span className="font-medium text-stone-800">
                        {getPlaasNaam(reenval.plaas_id)}
                      </span>
                    </TableCell>
                    <TableCell>
                      <div className="flex items-center gap-2">
                        <Calendar className="w-4 h-4 text-stone-500" />
                        {format(new Date(reenval.datum), 'dd MMM yyyy', { locale: af })}
                      </div>
                    </TableCell>
                    <TableCell>
                      <div className="flex items-center gap-2">
                        <Cloud className="w-4 h-4 text-blue-500" />
                        <span className="font-semibold text-blue-600">
                          {reenval.reenval_mm}mm
                        </span>
                      </div>
                    </TableCell>
                    <TableCell>
                      <Badge variant="secondary" className={tipeKleure[reenval.tipe_reenval]}>
                        {reenval.tipe_reenval.charAt(0).toUpperCase() + reenval.tipe_reenval.slice(1)}
                      </Badge>
                    </TableCell>
                    <TableCell>
                      <div className="flex gap-2">
                        <Button
                          size="sm"
                          variant="ghost"
                          onClick={() => onEdit(reenval)}
                          className="h-8 w-8 p-0 hover:bg-blue-100"
                        >
                          <Edit className="w-4 h-4 text-blue-600" />
                        </Button>
                        <Button
                          size="sm"
                          variant="ghost"
                          onClick={() => onDelete(reenval)}
                          className="h-8 w-8 p-0 hover:bg-red-100"
                        >
                          <Trash2 className="w-4 h-4 text-red-600" />
                        </Button>
                      </div>
                    </TableCell>
                  </TableRow>
                ))}
              </TableBody>
            </Table>
          </div>
        ) : (
          <div className="text-center py-12 text-stone-500">
            <Cloud className="w-16 h-16 mx-auto mb-4 opacity-50" />
            <p className="text-lg font-medium mb-2">Geen reenval rekords nog nie</p>
            <p>Begin deur jou eerste reenval entry te maak</p>
          </div>
        )}
      </CardContent>
    </Card>
  );
}
